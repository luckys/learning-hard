### Estrategia de Reintento

```typescript
/**
 * Retroceso exponencial: Esperar mÃ¡s tiempo entre reintentos
 * 
 * Intento 1: Esperar 1s
 * Intento 2: Esperar 2s
 * Intento 3: Esperar 4s
 * Intento 4: Esperar 8s
 * 
 * Â¿Por quÃ©? Le da tiempo al servidor para recuperarse
 */
async function fetchWithRetry(
  url: string,
  maxRetries = 3,
  baseDelay = 1000
) {
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      return await fetch(url)
    } catch (error) {
      if (attempt === maxRetries - 1) throw error
      
      // Retroceso exponencial
      const delay = baseDelay * Math.pow(2, attempt)
      await new Promise(resolve => setTimeout(resolve, delay))
    }
  }
}
```

```tsx
interface PaginationProps {
  currentPage: number
  lastPage: number
  onPageChange: (page: number) => void
}

export function Pagination({ currentPage, lastPage, onPageChange }: PaginationProps) {
  const pages = Array.from({ length: lastPage }, (_, i) => i + 1)

  return (
    <nav className="flex items-center justify-center gap-2 mt-8">
      <button
        onClick={() => onPageChange(currentPage - 1)}
        disabled={currentPage === 1}
        className="px-3 py-2 bg-slate-800 rounded-md disabled:opacity-50"
      >
        Anterior
      </button>

      {pages.map((page) => (
        <button
          key={page}
          onClick={() => onPageChange(page)}
          className={`px-3 py-2 rounded-md ${
            page === currentPage
              ? 'bg-blue-600 text-white'
              : 'bg-slate-800 text-slate-300'
          }`}
        >
          {page}
        </button>
      ))}

      <button
        onClick={() => onPageChange(currentPage + 1)}
        disabled={currentPage === lastPage}
        className="px-3 py-2 bg-slate-800 rounded-md disabled:opacity-50"
      >
        Siguiente
      </button>
    </nav>
  )
}
```
