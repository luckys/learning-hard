## Breadcrumbs Component

Breadcrumbs show the user's location in the site hierarchy:

```
Home > Categories > Technical Support > How to learn React?
```

```tsx
// src/components/layout/Breadcrumbs.tsx

import { Link } from 'react-router-dom'

export interface Breadcrumb {
  label: string
  href: string
}

interface BreadcrumbsProps {
  items: Breadcrumb[]
}

/**
 * Breadcrumbs - Show navigation path
 * 
 * Helps users understand their location in the site.
 * Improves SEO (search engines love breadcrumbs).
 * 
 * Semantic HTML:
 * - <nav> with aria-label="Breadcrumb"
 * - <ol> ordered list (hierarchical)
 * - Last item has aria-current="page"
 */
export function Breadcrumbs({ items }: BreadcrumbsProps) {
  return (
    <nav aria-label="Breadcrumb" className="mb-6">
      <ol className="flex items-center gap-2 text-sm">
        {/* Home link (always first) */}
        <BreadcrumbItem href="/" label="Home" />

        {/* Dynamic items */}
        {items.map((item, index) => (
          <BreadcrumbItem
            key={item.href}
            href={item.href}
            label={item.label}
            isLast={index === items.length - 1}
          />
        ))}
      </ol>
    </nav>
  )
}

function BreadcrumbItem({
  href,
  label,
  isLast = false,
}: {
  href: string
  label: string
  isLast?: boolean
}) {
  return (
    <>
      <li>
        {isLast ? (
          /* Current page: not a link */
          <span
            className="text-slate-100 font-medium"
            aria-current="page"
          >
            {label}
          </span>
        ) : (
          /* Previous pages: links */
          <Link
            to={href}
            className="text-slate-400 hover:text-slate-200 transition-colors"
          >
            {label}
          </Link>
        )}
      </li>

      {/* Separator */}
      {!isLast && (
        <li className="text-slate-600" aria-hidden="true">
          /
        </li>
      )}
    </>
  )
}
```

## Back Button Component

Simple navigation helper:

```tsx
// src/components/ui/BackButton.tsx

import { useNavigate } from 'react-router-dom'

interface BackButtonProps {
  fallback?: string  // Fallback URL if no history
  label?: string
}

/**
 * BackButton - Navigate to previous page
 * 
 * Uses browser history when available.
 * Falls back to specified URL if no history.
 */
export function BackButton({ fallback = '/', label = 'Back' }: BackButtonProps) {
  const navigate = useNavigate()

  const handleClick = () => {
    // Check if there's history
    if (window.history.length > 2) {
      navigate(-1)  // Go back
    } else {
      navigate(fallback)  // Go to fallback
    }
  }

  return (
    <button
      onClick={handleClick}
      className="flex items-center gap-2 text-slate-400 hover:text-slate-200 transition-colors"
    >
      <span>‚Üê</span>
      <span>{label}</span>
    </button>
  )
}
```

## Scroll to Top

When navigating between pages, scroll to top:

```tsx
// src/components/layout/ScrollToTop.tsx

import { useEffect } from 'react'
import { useLocation } from 'react-router-dom'

/**
 * ScrollToTop - Scroll to top on route change
 * 
 * Place in App.tsx inside BrowserRouter.
 * Scrolls window to top whenever route changes.
 */
export function ScrollToTop() {
  const { pathname } = useLocation()

  useEffect(() => {
    window.scrollTo(0, 0)
  }, [pathname])

  return null  // Doesn't render anything
}

// Usage in App.tsx:
<BrowserRouter>
  <ScrollToTop />
  <Routes>...</Routes>
</BrowserRouter>
```

## Progress Bar

Show loading state during navigation:

```tsx
// src/components/layout/ProgressBar.tsx

import { useEffect, useState } from 'react'
import { useNavigation } from 'react-router-dom'

/**
 * ProgressBar - Loading indicator for navigation
 * 
 * Shows at top of page during navigation.
 * Uses React Router's useNavigation hook.
 */
export function ProgressBar() {
  const navigation = useNavigation()
  const [progress, setProgress] = useState(0)

  useEffect(() => {
    if (navigation.state === 'loading') {
      // Simulate progress
      setProgress(10)
      const timer1 = setTimeout(() => setProgress(40), 100)
      const timer2 = setTimeout(() => setProgress(70), 300)
      
      return () => {
        clearTimeout(timer1)
        clearTimeout(timer2)
      }
    } else {
      setProgress(100)
      setTimeout(() => setProgress(0), 200)
    }
  }, [navigation.state])

  if (progress === 0) return null

  return (
    <div className="fixed top-0 left-0 right-0 h-1 bg-slate-800 z-50">
      <div
        className="h-full bg-blue-600 transition-all duration-300"
        style={{ width: `${progress}%` }}
      />
    </div>
  )
}
```
