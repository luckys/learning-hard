### Retry Strategy

```typescript
/**
 * Exponential backoff: Wait longer between retries
 * 
 * Attempt 1: Wait 1s
 * Attempt 2: Wait 2s
 * Attempt 3: Wait 4s
 * Attempt 4: Wait 8s
 * 
 * Why? Gives server time to recover
 */
async function fetchWithRetry(
  url: string,
  maxRetries = 3,
  baseDelay = 1000
) {
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      return await fetch(url)
    } catch (error) {
      if (attempt === maxRetries - 1) throw error
      
      // Exponential backoff
      const delay = baseDelay * Math.pow(2, attempt)
      await new Promise(resolve => setTimeout(resolve, delay))
    }
  }
}
```

## AbortController: Cancel Requests

**Problem**: User navigates away, but request still pending. Wastes bandwidth, causes memory leaks.

**Solution**: AbortController to cancel requests.

```typescript
/**
 * AbortController: Cancel fetch requests
 * 
 * Use case: User navigates away before data loads
 * 
 * Without AbortController:
 * - Request continues in background
 * - Response updates unmounted component
 * - Memory leak warning
 * 
 * With AbortController:
 * - Request cancelled immediately
 * - No memory leak
 * - Clean cleanup
 */
useEffect(() => {
  const abortController = new AbortController()
  
  async function fetchThreads() {
    try {
      const response = await fetch('/api/threads', {
        signal: abortController.signal  // Pass abort signal
      })
      // ...
    } catch (error) {
      // Check if error is from abort
      if (error.name === 'AbortError') {
        console.log('Request cancelled')
        return  // Don't update state
      }
      // Handle other errors
    }
  }
  
  fetchThreads()
  
  // Cleanup: Cancel request if component unmounts
  return () => abortController.abort()
}, [])
```

## Fetching with useEffect

```tsx
// src/pages/ThreadsPage.tsx

import { useState, useEffect } from 'react'
import { apiClient } from '@/lib/api-client'
import type { Thread } from '@/types'

export function ThreadsPage() {
  const [threads, setThreads] = useState<Thread[]>([])
  const [isLoading, setIsLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    async function fetchThreads() {
      try {
        setIsLoading(true)
        setError(null)

        const response = await apiClient.get<{ data: Thread[] }>('/threads')
        setThreads(response.data.data)
      } catch (err) {
        setError(err instanceof Error ? err.message : 'Failed to load threads')
      } finally {
        setIsLoading(false)
      }
    }

    fetchThreads()
  }, [])  // Empty array = run once on mount

  if (isLoading) {
    return <ThreadListSkeleton />
  }

  if (error) {
    return (
      <div className="text-center py-12">
        <p className="text-red-400">{error}</p>
      </div>
    )
  }

  return (
    <div className="container mx-auto px-4 py-8">
      <h1 className="text-3xl font-bold text-slate-100 mb-6">
        Forum Threads
      </h1>
      <ThreadList threads={threads} />
    </div>
  )
}
```

## Pagination Component

```tsx
// src/components/ui/Pagination.tsx

interface PaginationProps {
  currentPage: number
  lastPage: number
  onPageChange: (page: number) => void
}

export function Pagination({ currentPage, lastPage, onPageChange }: PaginationProps) {
  const pages = Array.from({ length: lastPage }, (_, i) => i + 1)

  return (
    <nav className="flex items-center justify-center gap-2 mt-8">
      <button
        onClick={() => onPageChange(currentPage - 1)}
        disabled={currentPage === 1}
        className="px-3 py-2 bg-slate-800 rounded-md disabled:opacity-50"
      >
        Previous
      </button>

      {pages.map((page) => (
        <button
          key={page}
          onClick={() => onPageChange(page)}
          className={`px-3 py-2 rounded-md ${
            page === currentPage
              ? 'bg-blue-600 text-white'
              : 'bg-slate-800 text-slate-300'
          }`}
        >
          {page}
        </button>
      ))}

      <button
        onClick={() => onPageChange(currentPage + 1)}
        disabled={currentPage === lastPage}
        className="px-3 py-2 bg-slate-800 rounded-md disabled:opacity-50"
      >
        Next
      </button>
    </nav>
  )
}
```

## Error Handling Best Practices

### 1. Classify Errors

```typescript
function getErrorMessage(error: unknown): string {
  if (error instanceof TypeError) {
    return 'Network error. Check your connection.'
  }
  
  if (error instanceof Response) {
    switch (error.status) {
      case 401:
        return 'Session expired. Please sign in again.'
      case 403:
        return 'You don\'t have permission to view this.'
      case 404:
        return 'Not found.'
      case 429:
        return 'Too many requests. Please wait.'
      case 500:
        return 'Server error. Please try again later.'
      default:
        return 'Something went wrong. Please try again.'
    }
  }
  
  return 'Unknown error occurred.'
}
```

```typescript
const TIMEOUT_MS = 10000  // 10 seconds

const fetchWithTimeout = (url: string, timeout = TIMEOUT_MS) => {
```

## Summary

### Error Handling Patterns

1. **Multi-layer error handling**
   - Network layer (fetch)
   - Component layer (try/catch)
   - User layer (error message)

2. **Error classification**
   - Network errors â†’ Retry
   - 4xx errors â†’ Don't retry
   - 5xx errors â†’ Retry with backoff
   - Timeouts â†’ Retry

3. **AbortController**
   - Cancel requests on unmount
   - Prevent memory leaks
   - Clean up properly

4. **Retry strategy**
   - Exponential backoff
   - Max retry limit
   - Jitter to prevent thundering herd

5. **User feedback**
   - Clear error messages
   - Retry buttons
   - Graceful degradation

### Best Practices

- âœ… Always handle errors (don't ignore them)
- âœ… Classify errors (different handling for different types)
- âœ… Implement retry logic (with exponential backoff)
- âœ… Use AbortController (prevent memory leaks)
- âœ… Provide user feedback (helpful error messages)
- âœ… Test error scenarios (network failures, timeouts)
- âŒ Don't show technical errors to users
- âŒ Don't retry non-retryable errors (4xx)
- âŒ Don't ignore AbortError
- âŒ Don't update unmounted components
